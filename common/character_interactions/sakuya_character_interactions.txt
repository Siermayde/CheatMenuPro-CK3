sakuya_mark_character_interaction = {
	interface_priority = 300
	category = interaction_debug_main
	use_diplomatic_range = no
	ignores_pending_interaction_block = yes
	common_interaction = yes

	is_shown = {
		scope:actor = {
			is_ai = no
		}
	}

	on_accept = {
		if = {
			limit = {
				exists = global_var:sakuya_marked_character
			}
			global_var:sakuya_marked_character = {
				remove_character_flag = sakuya_marked_character
			}
			remove_global_variable = sakuya_marked_character
		}

		scope:recipient = {
			add_character_flag = sakuya_marked_character
			set_global_variable = {
				name = sakuya_marked_character
				value = this
			}
			if = {
				limit = {
					exists = primary_spouse
					primary_spouse = { is_alive = yes }
				}
				primary_spouse = {
					set_global_variable = {
						name = sakuya_p2_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = diarch
					diarch = { is_alive = yes }
				}
				diarch = {
					set_global_variable = {
						name = sakuya_p2_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = player_heir
					player_heir = { is_alive = yes }
				}
				player_heir = {
					set_global_variable = {
						name = sakuya_p2_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = liege
					liege = { is_alive = yes }
				}
				liege = {
					set_global_variable = {
						name = sakuya_p2_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = father
					father = { is_alive = yes }
				}
				father = {
					set_global_variable = {
						name = sakuya_p2_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = mother
					mother = { is_alive = yes }
				}
				mother = {
					set_global_variable = {
						name = sakuya_p2_character
						value = this
					}
				}
			}
			else = {
				set_global_variable = {
					name = sakuya_p2_character
					value = this
				}
			}

			if = {
				limit = {
					is_lowborn = yes
				}
				set_global_variable = {
					name = sakuya_p3_character
					value = this
				}
			}
			else_if = {
				limit = {
					exists = house.house_head
					house.house_head = {
						is_alive = yes
						this != scope:recipient
						this != scope:actor
					}
				}
				house.house_head = {
					set_global_variable = {
						name = sakuya_p3_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = dynasty.dynast
					dynasty.dynast = {
						is_alive = yes
						this != scope:recipient
						this != scope:actor
					}
				}
				dynasty.dynast = {
					set_global_variable = {
						name = sakuya_p3_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = father
					father = {
						is_alive = yes
						this != scope:recipient
						this != scope:actor
						OR = {
							NOT = { exists = global_var:sakuya_p2_character }
							this != global_var:sakuya_p2_character
						}
					}
				}
				father = {
					set_global_variable = {
						name = sakuya_p3_character
						value = this
					}
				}
			}
			else_if = {
				limit = {
					exists = father.father
					father.father = {
						is_alive = yes
						this != scope:recipient
						this != scope:actor
					}
				}
				father.father = {
					set_global_variable = {
						name = sakuya_p3_character
						value = this
					}
				}
			}
			else = {
				set_global_variable = {
					name = sakuya_p3_character
					value = this
				}
			}
		}
	}

	auto_accept = yes
}
